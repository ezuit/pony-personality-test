        } else {
            nextBtn.innerText = '下一题 🌟';
        }
    }

    function goToPrev() {
        if (currentIndex > 0) {
            currentIndex--;
            renderCurrent();
            updateProgress();
        }
    }

    function goToNext() {
        if (userAnswers[currentIndex] === null) {
            alert('🌸 请先选择答案再继续下一题哟～');
            return;
        }
        if (currentIndex === 35) {
            computeResult();
            return;
        }
        currentIndex++;
        renderCurrent();
        updateProgress();
    }

    function resetAll() {
        if (confirm('确定要重置所有答案吗？进度将会丢失。')) {
            userAnswers.fill(null);
            currentIndex = 0;
            resultDiv.style.display = 'none';
            resultDiv.innerHTML = '';
            renderCurrent();
            updateProgress();
        }
    }

    // 计算各维度总分并确定类型
    function computeResult() {
        // 检查完整性
        for (let i = 0; i < 36; i++) {
            if (userAnswers[i] === null) {
                alert(`还有第${i+1}题未回答，请返回完成~`);
                currentIndex = i;
                renderCurrent();
                updateProgress();
                return;
            }
        }
        // 累加维度分数
        let scores = { sweet:0, charming:0, natural:0, tsundere:0, gentle:0, mystic:0 };
        for (let i = 0; i < 36; i++) {
            const dim = questionsData[i].dim;
            scores[dim] += userAnswers[i];
        }
        // 找出最高分维度
        let maxDim = null, maxScore = -1;
        for (let d of DIMENSION_KEYS) {
            if (scores[d] > maxScore) {
                maxScore = scores[d];
                maxDim = d;
            }

        const typeKey = maxDim;
        const typeInfo = TYPES[typeKey];
        // 总分范围 36~180，显示参考
        const totalScore = userAnswers.reduce((a,b)=>a+b,0);
